// ========== MAJU/MUNDUR HARI (DENGAN MENYIMPAN HISTORY) ========== function advanceDays(delta) { let state = modes[currentMode]; let newDay = state.day + delta; // Batasan hari if(newDay < 0) newDay = 0; if(newDay > 150) newDay = 150; if(newDay === state.day) return; // Jika mundur, ambil dari history yang sudah ada if(delta < 0 && state.history && state.history[newDay] !== undefined) { state.height = state.history[newDay]; state.day = newDay; // Reset status mati karena mundur ke hari sebelumnya state.isDead = false; state.deadCause = null; state.deathType = null; state.wiltLevel = 0; updateUI(); updateChart(); drawSunflower(); return; } // Jika maju atau mundur ke hari yang belum ada di history for(let i = 0; i < Math.abs(delta); i++) { if(delta > 0) { // MAJU if(state.isDead) { state.day++; state.height = Math.max(0, state.height - 2); state.history.push(state.height); continue; } if(state.height >= MAX_HEIGHT) { state.day++; state.history.push(state.height); updateDeathAndWilt(state); continue; } let inc = dailyGrowth(state); state.height = parseFloat((state.height + inc).toFixed(2)); if(state.height > MAX_HEIGHT) state.height = MAX_HEIGHT; updateDeathAndWilt(state); state.day++; state.history.push(state.height); if(state.history.length > 400) state.history.shift(); } else { // MUNDUR if(state.day > 0) { state.day--; if(state.history[state.day] !== undefined) { state.height = state.history[state.day]; } // Reset status mati saat mundur state.isDead = false; state.deadCause = null; state.deathType = null; state.wiltLevel = 0; } } } updateUI(); drawSunflower(); updateChart(); } function changeDay(delta) { advanceDays(delta); }
