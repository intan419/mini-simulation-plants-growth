// ========== FUNGSI RE-SIMULASI DARI HARI TERTENTU ========== function recalculateFromDay(targetDay) { let state = modes[currentMode]; // Simulasi dari awal (hari 0) sampai targetDay let newHistory = [0]; let tempHeight = 0; let tempIsDead = false; let tempWilt = 0; let tempDeadCause = null; let tempDeathType = null; for(let d = 1; d <= targetDay; d++) { if(tempIsDead) { tempHeight = Math.max(0, tempHeight - 2); newHistory.push(tempHeight); continue; } let target = getTargetHeight(d); let diff = Math.max(0, target - tempHeight); let mockState = { isDead: false, height: tempHeight, wiltLevel: tempWilt }; let growth = diff * 0.32 * getGrowthFactor(mockState); growth = Math.min(6, Math.max(0, growth)); tempHeight = parseFloat((tempHeight + growth).toFixed(2)); if(tempHeight > MAX_HEIGHT) tempHeight = MAX_HEIGHT; if(d > 90) tempHeight = Math.max(0, tempHeight - 1.5); // Update stres & layu let stress = calculateStressLevel(); if(stress.level > 75) tempWilt = Math.min(100, tempWilt + 7); else if(stress.level > 55) tempWilt = Math.min(90, tempWilt + 4); else if(stress.level > 35) tempWilt = Math.min(75, tempWilt + 2); else tempWilt = Math.max(0, tempWilt - 2); if(d > 90) tempWilt = Math.min(100, tempWilt + 1.5); // Cek kematian let deathMock = { day: d, height: tempHeight, isDead: false }; let death = getDeathCause(deathMock); if(death || tempWilt >= 92) { tempIsDead = true; tempDeadCause = death || { type:"old", cause:"🍂 MATI USIA TUA", detail:"Siklus hidup selesai." }; tempDeathType = tempDeadCause.type; tempWilt = 100; } newHistory.push(tempHeight); } // Update state state.day = targetDay; state.height = tempHeight; state.history = newHistory; state.isDead = tempIsDead; state.deadCause = tempDeadCause; state.deathType = tempDeathType; state.wiltLevel = tempWilt; return state; } // ========== CHANGE DAY (TANPA HARUS RESET) ========== function changeDay(delta) { let state = modes[currentMode]; let newDay = state.day + delta; if(newDay < 0) newDay = 0; if(newDay > 150) newDay = 150; if(newDay === state.day) return; recalculateFromDay(newDay); updateUI(); updateChart(); drawSunflower(); } // ========== RESET FUNCTIONS (TIDAK DIUBAH) ========== function resetToOptimal() { env.water=100; env.light=80; env.temp=27; env.ph=6.5; env.humidity=60; env.fertilizer="npk"; env.fertDose=50; env.season="optimal"; document.getElementById("water").value=100; document.getElementById("waterVal").innerText=100; document.getElementById("light").value=80; document.getElementById("lightVal").innerText=80; document.getElementById("temp").value=27; document.getElementById("tempVal").innerText=27; document.getElementById("ph").value=6.5; document.getElementById("phVal").innerText=6.5; document.getElementById("humidity").value=60; document.getElementById("humVal").innerText=60; document.getElementById("fertilizer").value="npk"; document.getElementById("fertDose").value=50; document.getElementById("fertVal").innerText=50; document.getElementById("season").value="optimal"; } function resetCurrentMode() { modes[currentMode] = { day:0, height:0, history:[0], isDead:false, deadCause:null, deathType:null, wiltLevel:0 }; if(currentMode === "normal") resetToOptimal(); recalculateFromDay(0); updateUI(); updateChart(); drawSunflower(); } function resetPreviousMode() { let prev = currentMode === "normal" ? "experiment" : "normal"; let oldMode = currentMode; currentMode = prev; modes[prev] = { day:0, height:0, history:[0], isDead:false, deadCause:null, deathType:null, wiltLevel:0 }; recalculateFromDay(0); currentMode = oldMode; updateChart(); if(currentMode === prev) { updateUI(); drawSunflower(); } } function resetAllModes() { modes.normal = { day:0, height:0, history:[0], isDead:false, deadCause:null, deathType:null, wiltLevel:0 }; modes.experiment = { day:0, height:0, history:[0], isDead:false, deadCause:null, deathType:null, wiltLevel:0 }; resetToOptimal(); let oldMode = currentMode; currentMode = "normal"; recalculateFromDay(0); currentMode = "experiment"; recalculateFromDay(0); currentMode = oldMode; updateUI(); updateChart(); drawSunflower(); }
